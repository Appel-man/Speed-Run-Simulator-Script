local Library = loadstring(game:HttpGet("https://pastebin.com/raw/yzt3qebK"))()
local Window  = Library.CreateLib("Speed Run SImulator Script", "Synapse")
local Player  = game.Players.LocalPlayer
-- MAIN
local Main        = Window:NewTab("Main")
local MainSection = Main:NewSection("Main")

local goToDb   = false
local goToLoop = false

function goTo()
    if goToDb and #workspace.OrbSpawns:GetChildren() > 0 then return end
    goToDb = true
    
    if workspace.OrbSpawns:FindFirstChild("Ring") then
        Player.Character:MoveTo(workspace.OrbSpawns:FindFirstChild("Ring").Position)
    else
        Player.Character:MoveTo(workspace.OrbSpawns:GetChildren()[1].Position)
    end
    goToDb = false
    return true
end

MainSection:NewButton("TP Orb", "Teleports You To All Available Orbs", function()
    goTo()
end)

MainSection:NewToggle("Auto Farm", "Loops The 'TP Orbs' Function", function(state)
    if state then
        goToLoop = true
        spawn(function()
            while goToLoop do
                repeat wait() until goTo()
                    wait(0.2)
            end
        end)
    else
        goToLoop = false
    end
end)

MainSection:NewButton("Inf Yield Admin", "FE Admin Commands", function()
    loadstring(game:HttpGet(('https://raw.githubusercontent.com/EdgeIY/infiniteyield/master/source'),true))()
end)
